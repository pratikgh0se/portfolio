MuleSoft DataWeave Transformation

// DataWeave transformation for enterprise data mapping
%dw 2.0
output application/json
import * from dw::core::Strings
import * from dw::core::Arrays

// Configuration for transformation rules
var config = {
    defaultTimezone: "UTC",
    dateFormat: "yyyy-MM-dd'T'HH:mm:ss'Z'",
    customerIdPrefix: "CUST_",
    defaultCountry: "US"
}

// Helper functions for data validation and transformation
fun validateEmail(email: String): Boolean = 
    email matches /^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$/

fun formatPhoneNumber(phone: String): String = 
    phone replace /[^\d]/ with "" 
    then ((number) -> if (sizeOf(number) == 10) "+1" ++ number else number)

fun normalizeAddress(address: Object): Object = {
    street: upper(address.street default ""),
    city: capitalize(address.city default ""),
    state: upper(address.state default ""),
    zipCode: address.zipCode default "",
    country: upper(address.country default config.defaultCountry)
}

// Main transformation logic
---
{
    // Customer information transformation
    customer: {
        id: config.customerIdPrefix ++ payload.customer.customerId,
        profile: {
            firstName: capitalize(payload.customer.firstName default ""),
            lastName: capitalize(payload.customer.lastName default ""),
            email: if (validateEmail(payload.customer.email)) 
                      payload.customer.email 
                   else null,
            phone: formatPhoneNumber(payload.customer.phone default ""),
            dateOfBirth: payload.customer.dob as Date {format: "MM/dd/yyyy"} 
                        as String {format: config.dateFormat},
            registrationDate: now() as String {format: config.dateFormat}
        },
        
        // Address normalization with validation
        addresses: payload.customer.addresses map ((address, index) -> {
            id: index + 1,
            type: upper(address.type default "BILLING"),
            ...(normalizeAddress(address)),
            isPrimary: index == 0,
            isValid: not isEmpty(address.street) and not isEmpty(address.city)
        }),
        
        // Preferences and settings
        preferences: {
            communication: {
                email: payload.customer.preferences.emailOptIn default true,
                sms: payload.customer.preferences.smsOptIn default false,
                phone: payload.customer.preferences.phoneOptIn default false
            },
            language: upper(payload.customer.preferences.language default "EN"),
            timezone: payload.customer.preferences.timezone default config.defaultTimezone,
            currency: upper(payload.customer.preferences.currency default "USD")
        }
    },
    
    // Order processing with business rules
    orders: payload.orders map ((order, orderIndex) -> {
        orderId: "ORD_" ++ order.id,
        customerId: config.customerIdPrefix ++ payload.customer.customerId,
        orderDate: order.createdAt as Date {format: "yyyy-MM-dd HH:mm:ss"} 
                  as String {format: config.dateFormat},
        
        status: {
            current: upper(order.status default "PENDING"),
            lastUpdated: now() as String {format: config.dateFormat},
            history: order.statusHistory default [] map ((status) -> {
                status: upper(status.status),
                timestamp: status.timestamp as Date {format: "yyyy-MM-dd HH:mm:ss"} 
                          as String {format: config.dateFormat},
                reason: status.reason default ""
            })
        },
        
        // Line items with pricing calculations
        lineItems: order.items map ((item, itemIndex) -> {
            id: itemIndex + 1,
            productId: item.sku,
            productName: item.name,
            quantity: item.quantity as Number,
            unitPrice: item.price as Number,
            totalPrice: (item.quantity as Number) * (item.price as Number),
            
            // Tax calculations
            tax: {
                rate: item.taxRate default 0.0875,
                amount: ((item.quantity as Number) * (item.price as Number)) * 
                       (item.taxRate default 0.0875)
            },
            
            // Discount applications
            discounts: item.discounts default [] map ((discount) -> {
                code: discount.code,
                type: upper(discount.type),
                amount: discount.amount as Number,
                percentage: discount.percentage default 0
            })
        }),
        
        // Shipping and fulfillment
        shipping: {
            method: order.shipping.method default "STANDARD",
            cost: order.shipping.cost as Number default 0,
            estimatedDelivery: order.shipping.estimatedDelivery as Date {format: "yyyy-MM-dd"} 
                              as String {format: config.dateFormat},
            trackingNumber: order.shipping.trackingNumber default null,
            carrier: order.shipping.carrier default "FEDEX"
        },
        
        // Payment information (sanitized)
        payment: {
            method: order.payment.method,
            status: upper(order.payment.status default "PENDING"),
            transactionId: order.payment.transactionId,
            // Sensitive data removed for security
            lastFourDigits: order.payment.cardNumber[-4 to -1] default null,
            authorizationCode: order.payment.authCode default null
        },
        
        // Calculated totals
        totals: {
            subtotal: sum(order.items map ((item) -> 
                         (item.quantity as Number) * (item.price as Number))),
            taxTotal: sum(order.items map ((item) -> 
                         ((item.quantity as Number) * (item.price as Number)) * 
                         (item.taxRate default 0.0875))),
            shippingTotal: order.shipping.cost as Number default 0,
            discountTotal: sum(order.items..discounts map ((discount) -> 
                              discount.amount as Number default 0)),
            grandTotal: sum(order.items map ((item) -> 
                           (item.quantity as Number) * (item.price as Number))) +
                       sum(order.items map ((item) -> 
                           ((item.quantity as Number) * (item.price as Number)) * 
                           (item.taxRate default 0.0875))) +
                       (order.shipping.cost as Number default 0) -
                       sum(order.items..discounts map ((discount) -> 
                           discount.amount as Number default 0))
        }
    }),
    
    // Metadata for tracking and auditing
    metadata: {
        transformationVersion: "2.1.0",
        processedAt: now() as String {format: config.dateFormat},
        sourceSystem: payload.sourceSystem default "LEGACY_CRM",
        targetSystem: "ENTERPRISE_ERP",
        recordCount: {
            customers: 1,
            orders: sizeOf(payload.orders default []),
            lineItems: sum(payload.orders default [] map sizeOf($.items default []))
        },
        validationResults: {
            emailValid: validateEmail(payload.customer.email),
            addressComplete: not isEmpty(payload.customer.addresses[0].street default ""),
            phoneFormatted: not isEmpty(formatPhoneNumber(payload.customer.phone default ""))
        }
    }
}

Jenkins CI/CD Pipeline for API Integration

// Jenkins pipeline for automated API integration deployment
pipeline {
    agent {
        label 'mulesoft-agent'
    }
    
    parameters {
        choice(
            name: 'ENVIRONMENT',
            choices: ['dev', 'staging', 'production'],
            description: 'Target deployment environment'
        )
        string(
            name: 'API_VERSION',
            defaultValue: 'v1.0',
            description: 'API version to deploy'
        )
        booleanParam(
            name: 'RUN_INTEGRATION_TESTS',
            defaultValue: true,
            description: 'Execute integration test suite'
        )
        booleanParam(
            name: 'DEPLOY_TO_RUNTIME',
            defaultValue: true,
            description: 'Deploy to MuleSoft runtime'
        )
    }
    
    environment {
        MULE_HOME = '/opt/mule'
        ANYPOINT_USERNAME = credentials('anypoint-username')
        ANYPOINT_PASSWORD = credentials('anypoint-password')
        ANYPOINT_ORG_ID = credentials('anypoint-org-id')
        SLACK_CHANNEL = '#integrations'
    }
    
    stages {
        stage('Checkout and Setup') {
            steps {
                // Checkout integration code
                checkout scm
                
                script {
                    // Set dynamic environment variables
                    env.BUILD_VERSION = "${params.API_VERSION}-${BUILD_NUMBER}"
                    env.TARGET_ENV = params.ENVIRONMENT.toLowerCase()
                    
                    // Load environment-specific configuration
                    def configFile = "config/${env.TARGET_ENV}.properties"
                    if (fileExists(configFile)) {
                        def props = readProperties file: configFile
                        env.API_BASE_URL = props['api.base.url']
                        env.DATABASE_URL = props['database.url']
                        env.QUEUE_NAME = props['queue.name']
                    }
                }
                
                // Install dependencies
                sh """
                    # Install Maven dependencies
                    mvn clean install -DskipTests=true
                    
                    # Validate MuleSoft project structure
                    if [ ! -f "mule-artifact.json" ]; then
                        echo "Error: mule-artifact.json not found"
                        exit 1
                    fi
                """
            }
        }
        
        stage('Code Quality & Security') {
            parallel {
                stage('Static Code Analysis') {
                    steps {
                        sh """
                            # SonarQube analysis for MuleSoft
                            mvn sonar:sonar \
                                -Dsonar.host.url=${SONAR_HOST_URL} \
                                -Dsonar.login=${SONAR_TOKEN} \
                                -Dsonar.projectKey=api-integration-${params.ENVIRONMENT}
                        """
                    }
                }
                
                stage('Dependency Security Scan') {
                    steps {
                        sh """
                            # OWASP dependency check
                            mvn org.owasp:dependency-check-maven:check \
                                -DfailBuildOnCVSS=7 \
                                -DsuppressionsLocation=owasp-suppressions.xml
                        """
                    }
                }
                
                stage('DataWeave Unit Tests') {
                    steps {
                        sh """
                            # Run DataWeave transformation tests
                            mvn test \
                                -Dtest=TransformationTest \
                                -Dmaven.test.failure.ignore=false
                        """
                        
                        publishTestResults testResultsPattern: 'target/surefire-reports/*.xml'
                    }
                }
            }
        }
        
        stage('Build and Package') {
            steps {
                sh """
                    # Build MuleSoft application
                    mvn clean package \
                        -DskipTests=false \
                        -Dmule.env=${params.ENVIRONMENT} \
                        -Dapi.version=${params.API_VERSION}
                    
                    # Verify build artifacts
                    if [ ! -f "target/*.jar" ]; then
                        echo "Error: Build artifact not found"
                        exit 1
                    fi
                    
                    # Create deployment package
                    mkdir -p deployment
                    cp target/*.jar deployment/
                    cp -r src/main/resources/config/${params.ENVIRONMENT}/* deployment/ || true
                """
                
                archiveArtifacts artifacts: 'target/*.jar, deployment/*', fingerprint: true
            }
        }
        
        stage('Integration Testing') {
            when {
                expression { params.RUN_INTEGRATION_TESTS }
            }
            steps {
                script {
                    // Deploy to test environment first
                    sh """
                        # Deploy to test runtime
                        curl -X POST \
                            -H "Content-Type: application/json" \
                            -H "Authorization: Bearer \$(curl -s -X POST https://anypoint.mulesoft.com/accounts/login \
                                -H "Content-Type: application/json" \
                                -d '{"username":"'${ANYPOINT_USERNAME}'","password":"'${ANYPOINT_PASSWORD}'"}' | \
                                jq -r '.access_token')" \
                            -d '{
                                "artifactName": "api-integration-${BUILD_VERSION}",
                                "file": "@target/api-integration-${BUILD_VERSION}.jar"
                            }' \
                            "https://anypoint.mulesoft.com/cloudhub/api/v2/applications"
                    """
                    
                    // Wait for deployment
                    sleep(time: 60, unit: 'SECONDS')
                    
                    // Run API tests
                    sh """
                        # Run Postman/Newman integration tests
                        newman run tests/integration/api-integration-tests.json \
                            --environment tests/environments/${params.ENVIRONMENT}.json \
                            --reporters cli,junit \
                            --reporter-junit-export test-results.xml
                        
                        # Validate DataWeave transformations
                        python3 scripts/validate_transformations.py \
                            --environment ${params.ENVIRONMENT} \
                            --api-base-url ${env.API_BASE_URL}
                    """
                }
                
                publishTestResults testResultsPattern: 'test-results.xml'
            }
        }
        
        stage('Deploy to Runtime') {
            when {
                expression { params.DEPLOY_TO_RUNTIME }
            }
            steps {
                script {
                    def deploymentConfig = [
                        dev: [
                            workers: 1,
                            vCore: 0.1,
                            region: 'us-east-1'
                        ],
                        staging: [
                            workers: 2,
                            vCore: 0.2,
                            region: 'us-east-1'
                        ],
                        production: [
                            workers: 4,
                            vCore: 1.0,
                            region: 'us-east-1'
                        ]
                    ]
                    
                    def config = deploymentConfig[params.ENVIRONMENT]
                    
                    sh """
                        # Get Anypoint access token
                        ACCESS_TOKEN=\$(curl -s -X POST https://anypoint.mulesoft.com/accounts/login \
                            -H "Content-Type: application/json" \
                            -d '{"username":"'${ANYPOINT_USERNAME}'","password":"'${ANYPOINT_PASSWORD}'"}' | \
                            jq -r '.access_token')
                        
                        # Check if application exists
                        APP_EXISTS=\$(curl -s -H "Authorization: Bearer \$ACCESS_TOKEN" \
                            "https://anypoint.mulesoft.com/cloudhub/api/v2/applications/api-integration-${params.ENVIRONMENT}" | \
                            jq -r '.domain // empty')
                        
                        if [ -n "\$APP_EXISTS" ]; then
                            echo "Updating existing application..."
                            # Update existing application
                            curl -X PUT \
                                -H "Content-Type: application/json" \
                                -H "Authorization: Bearer \$ACCESS_TOKEN" \
                                -F "file=@target/api-integration-${BUILD_VERSION}.jar" \
                                -F 'appInfoJson={
                                    "workers": ${config.workers},
                                    "vCores": ${config.vCore},
                                    "region": "${config.region}",
                                    "properties": {
                                        "environment": "${params.ENVIRONMENT}",
                                        "api.version": "${params.API_VERSION}"
                                    }
                                };type=application/json' \
                                "https://anypoint.mulesoft.com/cloudhub/api/v2/applications/api-integration-${params.ENVIRONMENT}"
                        else
                            echo "Creating new application..."
                            # Create new application
                            curl -X POST \
                                -H "Content-Type: application/json" \
                                -H "Authorization: Bearer \$ACCESS_TOKEN" \
                                -F "file=@target/api-integration-${BUILD_VERSION}.jar" \
                                -F 'appInfoJson={
                                    "domain": "api-integration-${params.ENVIRONMENT}",
                                    "workers": ${config.workers},
                                    "vCores": ${config.vCore},
                                    "region": "${config.region}",
                                    "properties": {
                                        "environment": "${params.ENVIRONMENT}",
                                        "api.version": "${params.API_VERSION}"
                                    }
                                };type=application/json' \
                                "https://anypoint.mulesoft.com/cloudhub/api/v2/applications"
                        fi
                    """
                }
            }
        }
        
        stage('Post-Deployment Validation') {
            steps {
                script {
                    // Wait for application to start
                    timeout(time: 10, unit: 'MINUTES') {
                        waitUntil {
                            script {
                                def response = sh(
                                    script: """
                                        curl -s -o /dev/null -w '%{http_code}' \
                                            --max-time 30 \
                                            ${env.API_BASE_URL}/health
                                    """,
                                    returnStdout: true
                                ).trim()
                                
                                return response == '200'
                            }
                        }
                    }
                    
                    // Validate API endpoints
                    sh """
                        # Test critical API endpoints
                        echo "Testing API endpoints..."
                        
                        # Health check
                        curl -f "${env.API_BASE_URL}/health" || exit 1
                        
                        # API documentation
                        curl -f "${env.API_BASE_URL}/api/console" || exit 1
                        
                        # Sample data transformation
                        curl -X POST \
                            -H "Content-Type: application/json" \
                            -d '{"test": "data"}' \
                            "${env.API_BASE_URL}/api/transform" || exit 1
                        
                        echo "All endpoint tests passed"
                    """
                }
            }
        }
        
        stage('API Documentation Update') {
            steps {
                sh """
                    # Update API documentation in Anypoint Exchange
                    ACCESS_TOKEN=\$(curl -s -X POST https://anypoint.mulesoft.com/accounts/login \
                        -H "Content-Type: application/json" \
                        -d '{"username":"'${ANYPOINT_USERNAME}'","password":"'${ANYPOINT_PASSWORD}'"}' | \
                        jq -r '.access_token')
                    
                    # Upload RAML specification
                    if [ -f "src/main/resources/api/api.raml" ]; then
                        curl -X PUT \
                            -H "Authorization: Bearer \$ACCESS_TOKEN" \
                            -H "Content-Type: application/zip" \
                            --data-binary @api-spec.zip \
                            "https://anypoint.mulesoft.com/exchange/api/v1/assets/${ANYPOINT_ORG_ID}/api-integration/${params.API_VERSION}"
                    fi
                """
            }
        }
    }
    
    post {
        success {
            slackSend(
                channel: env.SLACK_CHANNEL,
                color: 'good',
                message: """
                    ✅ *API Integration Deployment Successful*
                    *Environment:* `${params.ENVIRONMENT}`
                    *Version:* `${params.API_VERSION}`
                    *Build:* `#${BUILD_NUMBER}`
                    *Duration:* ${currentBuild.durationString}
                """
            )
        }
        
        failure {
            slackSend(
                channel: env.SLACK_CHANNEL,
                color: 'danger',
                message: """
                    ❌ *API Integration Deployment Failed*
                    *Environment:* `${params.ENVIRONMENT}`
                    *Version:* `${params.API_VERSION}`
                    *Build Log:* <${BUILD_URL}console|View Details>
                """
            )
        }
        
        always {
            // Archive logs and reports
            archiveArtifacts artifacts: 'logs/*.log, reports/*.html', allowEmptyArchive: true
            
            // Clean workspace
            cleanWs()
        }
    }
}

Python Integration Testing Framework

# Python framework for comprehensive API integration testing
import requests
import json
import pytest
import time
import logging
from typing import Dict, List, Any, Optional
from dataclasses import dataclass
from datetime import datetime, timedelta
import xml.etree.ElementTree as ET

@dataclass
class APIEndpoint:
    name: str
    url: str
    method: str
    headers: Dict[str, str]
    payload: Optional[Dict[str, Any]] = None
    expected_status: int = 200
    timeout: int = 30

@dataclass
class TransformationTestCase:
    name: str
    input_data: Dict[str, Any]
    expected_output: Dict[str, Any]
    transformation_endpoint: str

class APIIntegrationTester:
    def __init__(self, base_url: str, environment: str):
        self.base_url = base_url.rstrip('/')
        self.environment = environment
        self.session = requests.Session()
        self.test_results = []
        
        # Configure logging
        logging.basicConfig(level=logging.INFO)
        self.logger = logging.getLogger(__name__)
        
        # Default headers
        self.session.headers.update({
            'Content-Type': 'application/json',
            'User-Agent': f'APIIntegrationTester/1.0 ({environment})',
            'X-Test-Environment': environment
        })
    
    def test_api_health_endpoints(self) -> Dict[str, bool]:
        """Test all health and status endpoints"""
        health_endpoints = [
            APIEndpoint(
                name="Health Check",
                url=f"{self.base_url}/health",
                method="GET",
                headers={}
            ),
            APIEndpoint(
                name="Ready Check",
                url=f"{self.base_url}/ready",
                method="GET",
                headers={}
            ),
            APIEndpoint(
                name="Metrics Endpoint",
                url=f"{self.base_url}/metrics",
                method="GET",
                headers={'Accept': 'text/plain'}
            )
        ]
        
        results = {}
        for endpoint in health_endpoints:
            try:
                response = self.session.request(
                    method=endpoint.method,
                    url=endpoint.url,
                    headers=endpoint.headers,
                    timeout=endpoint.timeout
                )
                
                success = response.status_code == endpoint.expected_status
                results[endpoint.name] = success
                
                self.logger.info(f"{endpoint.name}: {'PASS' if success else 'FAIL'} "
                               f"({response.status_code})")
                
            except Exception as e:
                results[endpoint.name] = False
                self.logger.error(f"{endpoint.name}: FAIL - {e}")
        
        return results
    
    def test_data_transformations(self, test_cases: List[TransformationTestCase]) -> Dict[str, bool]:
        """Test DataWeave transformations with various input scenarios"""
        results = {}
        
        for test_case in test_cases:
            try:
                self.logger.info(f"Testing transformation: {test_case.name}")
                
                # Send transformation request
                response = self.session.post(
                    url=f"{self.base_url}{test_case.transformation_endpoint}",
                    json=test_case.input_data,
                    timeout=30
                )
                
                if response.status_code != 200:
                    results[test_case.name] = False
                    self.logger.error(f"Transformation failed with status {response.status_code}")
                    continue
                
                # Parse response
                actual_output = response.json()
                
                # Validate transformation results
                validation_result = self._validate_transformation_output(
                    actual_output, 
                    test_case.expected_output
                )
                
                results[test_case.name] = validation_result
                self.logger.info(f"{test_case.name}: {'PASS' if validation_result else 'FAIL'}")
                
            except Exception as e:
                results[test_case.name] = False
                self.logger.error(f"{test_case.name}: FAIL - {e}")
        
        return results
    
    def _validate_transformation_output(self, actual: Dict, expected: Dict) -> bool:
        """Validate transformation output matches expected structure and values"""
        try:
            # Check required fields exist
            for key in expected.keys():
                if key not in actual:
                    self.logger.error(f"Missing required field: {key}")
                    return False
            
            # Validate customer data structure
            if 'customer' in expected:
                customer_valid = self._validate_customer_data(
                    actual.get('customer', {}), 
                    expected.get('customer', {})
                )
                if not customer_valid:
                    return False
            
            # Validate order data structure
            if 'orders' in expected:
                orders_valid = self._validate_orders_data(
                    actual.get('orders', []), 
                    expected.get('orders', [])
                )
                if not orders_valid:
                    return False
            
            # Validate metadata
            if 'metadata' in expected:
                metadata_valid = self._validate_metadata(
                    actual.get('metadata', {}), 
                    expected.get('metadata', {})
                )
                if not metadata_valid:
                    return False
            
            return True
            
        except Exception as e:
            self.logger.error(f"Validation error: {e}")
            return False
    
    def _validate_customer_data(self, actual: Dict, expected: Dict) -> bool:
        """Validate customer data transformation"""
        required_fields = ['id', 'profile', 'addresses', 'preferences']
        
        for field in required_fields:
            if field not in actual:
                self.logger.error(f"Missing customer field: {field}")
                return False
        
        # Validate profile data
        profile = actual.get('profile', {})
        required_profile_fields = ['firstName', 'lastName', 'email']
        
        for field in required_profile_fields:
            if field not in profile:
                self.logger.error(f"Missing profile field: {field}")
                return False
        
        # Validate email format
        email = profile.get('email')
        if email and '@' not in email:
            self.logger.error(f"Invalid email format: {email}")
            return False
        
        return True
    
    def _validate_orders_data(self, actual: List, expected: List) -> bool:
        """Validate orders data transformation"""
        if len(actual) != len(expected):
            self.logger.error(f"Order count mismatch: {len(actual)} vs {len(expected)}")
            return False
        
        for order in actual:
            required_fields = ['orderId', 'customerId', 'orderDate', 'status', 'lineItems', 'totals']
            
            for field in required_fields:
                if field not in order:
                    self.logger.error(f"Missing order field: {field}")
                    return False
            
            # Validate line items
            for item in order.get('lineItems', []):
                item_fields = ['id', 'productId', 'quantity', 'unitPrice', 'totalPrice']
                for field in item_fields:
                    if field not in item:
                        self.logger.error(f"Missing line item field: {field}")
                        return False
        
        return True
    
    def _validate_metadata(self, actual: Dict, expected: Dict) -> bool:
        """Validate metadata information"""
        required_fields = ['transformationVersion', 'processedAt', 'sourceSystem', 'targetSystem']
        
        for field in required_fields:
            if field not in actual:
                self.logger.error(f"Missing metadata field: {field}")
                return False
        
        return True
    
    def test_api_performance(self, endpoints: List[APIEndpoint], 
                           concurrent_requests: int = 10) -> Dict[str, Dict[str, float]]:
        """Test API performance under load"""
        import concurrent.futures
        import statistics
        
        results = {}
        
        for endpoint in endpoints:
            self.logger.info(f"Performance testing: {endpoint.name}")
            
            response_times = []
            success_count = 0
            
            def make_request():
                start_time = time.time()
                try:
                    response = self.session.request(
                        method=endpoint.method,
                        url=endpoint.url,
                        json=endpoint.payload,
                        headers=endpoint.headers,
                        timeout=endpoint.timeout
                    )
                    end_time = time.time()
                    
                    if response.status_code == endpoint.expected_status:
                        return end_time - start_time, True
                    else:
                        return end_time - start_time, False
                        
                except Exception:
                    end_time = time.time()
                    return end_time - start_time, False
            
            # Execute concurrent requests
            with concurrent.futures.ThreadPoolExecutor(max_workers=concurrent_requests) as executor:
                futures = [executor.submit(make_request) for _ in range(concurrent_requests)]
                
                for future in concurrent.futures.as_completed(futures):
                    response_time, success = future.result()
                    response_times.append(response_time)
                    if success:
                        success_count += 1
            
            # Calculate performance metrics
            results[endpoint.name] = {
                'avg_response_time': statistics.mean(response_times),
                'min_response_time': min(response_times),
                'max_response_time': max(response_times),
                'p95_response_time': statistics.quantiles(response_times, n=20)[18],
                'success_rate': success_count / concurrent_requests,
                'total_requests': concurrent_requests
            }
            
            self.logger.info(f"{endpoint.name} performance: "
                           f"avg={results[endpoint.name]['avg_response_time']:.3f}s, "
                           f"success_rate={results[endpoint.name]['success_rate']:.2%}")
        
        return results
    
    def generate_test_report(self, health_results: Dict, transformation_results: Dict, 
                           performance_results: Dict) -> Dict[str, Any]:
        """Generate comprehensive test report"""
        total_tests = len(health_results) + len(transformation_results)
        passed_tests = sum([health_results.values()]) + sum([transformation_results.values()])
        
        report = {
            'test_summary': {
                'total_tests': total_tests,
                'passed_tests': passed_tests,
                'failed_tests': total_tests - passed_tests,
                'success_rate': passed_tests / total_tests if total_tests > 0 else 0,
                'environment': self.environment,
                'timestamp': datetime.utcnow().isoformat()
            },
            'health_check_results': health_results,
            'transformation_results': transformation_results,
            'performance_results': performance_results
        }
        
        return report
