Project Overview

Designed and implemented a comprehensive decentralized feature flag system at Highspot, enabling safe deployment patterns, enhanced development velocity, and improved operational control. Challenge

Highspot's development teams needed a robust feature flag system to enable safe deployment patterns, A/B testing capabilities, and gradual feature rollouts. The existing manual configuration approach was error-prone, lacked centralized management, and hindered rapid development cycles across multiple microservices architectures.

LaunchDarkly Integration (Clojure)

;; Clojure implementation for LaunchDarkly feature flag integration
(ns feature-flags.core
  (:require [clojure.tools.logging :as log]
            [environ.core :refer [env]]
            [cheshire.core :as json])
  (:import [com.launchdarkly.sdk LDUser LDValue]
           [com.launchdarkly.sdk.server LDClient LDConfig]))

(defn create-ld-config
  "Create LaunchDarkly configuration with enhanced settings"
  [sdk-key]
  (-> (LDConfig/builder)
      (.sdkKey sdk-key)
      (.offline false)
      (.allAttributesPrivate true)
      (.events true)
      (.diagnosticOptOut false)
      (.build)))

(defn create-user-context
  "Create LaunchDarkly user context with team and environment attributes"
  [user-id team environment]
  (-> (LDUser/builder user-id)
      (.custom "team" team)
      (.custom "environment" environment)
      (.custom "service" (env :service-name "unknown"))
      (.build)))

(defprotocol FeatureFlagService
  (get-flag [this flag-key user-context default-value])
  (get-json-flag [this flag-key user-context default-value])
  (all-flags [this user-context])
  (close [this]))

(defrecord LaunchDarklyService [^LDClient client]
  FeatureFlagService
  
  (get-flag [this flag-key user-context default-value]
    (try
      (let [flag-value (.boolVariation client flag-key user-context default-value)]
        (log/debug "Feature flag" flag-key "evaluated to" flag-value "for user" (.getKey user-context))
        flag-value)
      (catch Exception e
        (log/error e "Failed to evaluate feature flag" flag-key)
        default-value)))
  
  (get-json-flag [this flag-key user-context default-value]
    (try
      (let [json-value (.jsonValueVariation client flag-key user-context (LDValue/of default-value))
            parsed-value (json/parse-string (.toJsonString json-value) true)]
        (log/debug "JSON feature flag" flag-key "evaluated for user" (.getKey user-context))
        parsed-value)
      (catch Exception e
        (log/error e "Failed to evaluate JSON feature flag" flag-key)
        default-value)))
  
  (all-flags [this user-context]
    (try
      (let [flags (.allFlagsState client user-context)]
        (into {} (map (fn [[k v]] [k (.getValue v)]) (.asMap flags))))
      (catch Exception e
        (log/error e "Failed to get all flags for user" (.getKey user-context))
        {})))
  
  (close [this]
    (when client
      (.close client))))

(defn create-feature-flag-service
  "Initialize LaunchDarkly service with configuration"
  []
  (let [sdk-key (env :launchdarkly-sdk-key)]
    (if sdk-key
      (do
        (log/info "Initializing LaunchDarkly service")
        (let [config (create-ld-config sdk-key)
              client (LDClient. config)]
          (if (.isInitialized client)
            (do
              (log/info "LaunchDarkly client initialized successfully")
              (->LaunchDarklyService client))
            (do
              (log/warn "LaunchDarkly client failed to initialize, using offline mode")
              nil))))
      (do
        (log/warn "LaunchDarkly SDK key not found, feature flags disabled")
        nil))))

;; Feature flag evaluation helpers
(defn evaluate-feature-flag
  "Evaluate a feature flag with user context"
  [flag-service flag-key {:keys [user-id team environment]} default-value]
  (if flag-service
    (let [user-context (create-user-context user-id team environment)]
      (get-flag flag-service flag-key user-context default-value))
    default-value))

(defn with-feature-flag
  "Conditional execution based on feature flag"
  [flag-service flag-key user-context then-fn else-fn]
  (if (evaluate-feature-flag flag-service flag-key user-context false)
    (then-fn)
    (else-fn)))

;; Example usage in service layer
(defn process-user-request
  "Example service function using feature flags"
  [flag-service request]
  (let [user-context {:user-id (:user-id request)
                     :team (:team request)
                     :environment "production"}]
    
    ;; Example: New authentication flow
    (if (evaluate-feature-flag flag-service "new-auth-flow" user-context false)
      (do
        (log/info "Using new authentication flow for user" (:user-id request))
        (new-authentication-handler request))
      (do
        (log/info "Using legacy authentication flow for user" (:user-id request))
        (legacy-authentication-handler request)))
    
    ;; Example: Dynamic configuration
    (let [cache-config (evaluate-feature-flag 
                        flag-service 
                        "cache-configuration" 
                        user-context 
                        {:ttl 3600 :max-size 1000})]
      (configure-cache cache-config))
    
    ;; Example: Progressive feature rollout
    (when (evaluate-feature-flag flag-service "analytics-v2" user-context false)
      (send-analytics-v2 request))))

# Kubernetes ESO configuration for secure LaunchDarkly token management
apiVersion: external-secrets.io/v1beta1
kind: SecretStore
metadata:
  name: aws-secrets-manager
  namespace: feature-flags
spec:
  provider:
    aws:
      service: SecretsManager
      region: us-west-2
      auth:
        jwt:
          serviceAccountRef:
            name: external-secrets-sa

---
# External Secret for LaunchDarkly SDK keys
apiVersion: external-secrets.io/v1beta1
kind: ExternalSecret
metadata:
  name: launchdarkly-secrets
  namespace: feature-flags
spec:
  refreshInterval: 300s  # 5 minutes
  secretStoreRef:
    name: aws-secrets-manager
    kind: SecretStore
  
  target:
    name: launchdarkly-sdk-keys
    creationPolicy: Owner
    template:
      type: Opaque
      data:
        production-sdk-key: "{{ .productionSdkKey }}"
        staging-sdk-key: "{{ .stagingSdkKey }}"
        mobile-sdk-key: "{{ .mobileSdkKey }}"
  
  data:
  - secretKey: productionSdkKey
    remoteRef:
      key: feature-flags/launchdarkly
      property: production_sdk_key
  - secretKey: stagingSdkKey
    remoteRef:
      key: feature-flags/launchdarkly
      property: staging_sdk_key
  - secretKey: mobileSdkKey
    remoteRef:
      key: feature-flags/launchdarkly
      property: mobile_sdk_key

---
# ClusterSecretStore for organization-wide access
apiVersion: external-secrets.io/v1beta1
kind: ClusterSecretStore
metadata:
  name: aws-secrets-manager-cluster
spec:
  provider:
    aws:
      service: SecretsManager
      region: us-west-2
      auth:
        jwt:
          serviceAccountRef:
            name: external-secrets-sa
            namespace: external-secrets-system

# Kubernetes deployment with LaunchDarkly integration
apiVersion: apps/v1
kind: Deployment
metadata:
  name: user-service
  namespace: production
  labels:
    app: user-service
    version: v2.1.0
spec:
  replicas: 3
  selector:
    matchLabels:
      app: user-service
  template:
    metadata:
      labels:
        app: user-service
        version: v2.1.0
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "9090"
        prometheus.io/path: "/metrics"
    spec:
      serviceAccountName: user-service-sa
      
      # Security context
      securityContext:
        runAsNonRoot: true
        runAsUser: 1000
        fsGroup: 2000
        
      containers:
      - name: user-service
        image: highspot/user-service:v2.1.0
        ports:
        - containerPort: 8080
          name: http
        - containerPort: 9090
          name: metrics
          
        # Environment variables for feature flag configuration
        env:
        - name: SERVICE_NAME
          value: "user-service"
        - name: ENVIRONMENT
          value: "production"
        - name: LAUNCHDARKLY_SDK_KEY
          valueFrom:
            secretKeyRef:
              name: launchdarkly-sdk-keys
              key: production-sdk-key
        - name: FEATURE_FLAGS_ENABLED
          value: "true"
        - name: LOG_LEVEL
          value: "info"
        - name: METRICS_ENABLED
          value: "true"
        
        # Health checks
        livenessProbe:
          httpGet:
            path: /health/live
            port: 8080
          initialDelaySeconds: 30
          periodSeconds: 10
          timeoutSeconds: 5
          failureThreshold: 3
          
        readinessProbe:
          httpGet:
            path: /health/ready
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 5
          timeoutSeconds: 3
          failureThreshold: 3
          
        # Resource limits
        resources:
          limits:
            cpu: 1000m
            memory: 1Gi
          requests:
            cpu: 500m
            memory: 512Mi
            
        # Volume mounts for configuration
        volumeMounts:
        - name: config
          mountPath: /app/config
          readOnly: true
      
      volumes:
      - name: config
        configMap:
          name: user-service-config
          
      # Pod disruption budget for high availability
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
          - weight: 100
            podAffinityTerm:
              labelSelector:
                matchExpressions:
                - key: app
                  operator: In
                  values:
                  - user-service
              topologyKey: kubernetes.io/hostname

Terraform AWS Infrastructure

# Terraform configuration for LaunchDarkly infrastructure
resource "aws_secretsmanager_secret" "launchdarkly_keys" {
  name        = "feature-flags/launchdarkly"
  description = "LaunchDarkly SDK keys for feature flag system"
  
  replica {
    region = "us-east-1"
  }
  
  tags = {
    Environment = var.environment
    Service     = "feature-flags"
    Purpose     = "launchdarkly-integration"
  }
}

resource "aws_secretsmanager_secret_version" "launchdarkly_keys" {
  secret_id = aws_secretsmanager_secret.launchdarkly_keys.id
  secret_string = jsonencode({
    production_sdk_key = var.launchdarkly_production_key
    staging_sdk_key    = var.launchdarkly_staging_key
    mobile_sdk_key     = var.launchdarkly_mobile_key
  })
  
  lifecycle {
    ignore_changes = [secret_string]
  }
}

# IAM role for External Secrets Operator
resource "aws_iam_role" "external_secrets_role" {
  name = "${var.environment}-external-secrets-role"
  
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Action = "sts:AssumeRoleWithWebIdentity"
        Effect = "Allow"
        Principal = {
          Federated = var.oidc_provider_arn
        }
        Condition = {
          StringEquals = {
            "${var.oidc_provider}:sub": "system:serviceaccount:external-secrets-system:external-secrets-sa"
            "${var.oidc_provider}:aud": "sts.amazonaws.com"
          }
        }
      }
    ]
  })
  
  tags = {
    Environment = var.environment
    Service     = "external-secrets"
  }
}

resource "aws_iam_policy" "external_secrets_policy" {
  name = "${var.environment}-external-secrets-policy"
  
  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Action = [
          "secretsmanager:GetSecretValue",
          "secretsmanager:DescribeSecret"
        ]
        Resource = [
          aws_secretsmanager_secret.launchdarkly_keys.arn,
          "${aws_secretsmanager_secret.launchdarkly_keys.arn}*"
        ]
      }
    ]
  })
}

resource "aws_iam_role_policy_attachment" "external_secrets_policy" {
  role       = aws_iam_role.external_secrets_role.name
  policy_arn = aws_iam_policy.external_secrets_policy.arn
}
