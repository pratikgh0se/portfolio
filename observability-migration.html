Project Overview

Led a monumental initiative to migrate Highspot's legacy observability systems from GCP/GKE to AWS and optimize metric collection, logging, and alerting. This project resulted in significant cost savings, improved system health visibility, and streamlined incident management through strategic consolidation and modern tooling adoption.

Situation

Highspot's observability infrastructure was a complex, fragmented landscape spanning multiple legacy tools (Prometheus v2, Thanos for long-term storage, self-hosted Grafana instances, Elasticsearch for logs, Rollbar for error tracking) and cloud providers (GCP/GKE and AWS). This multi-vendor, multi-cloud sprawl led to exorbitant operational costs (daily metric ingestion ~20TB, monthly storage ~256TB), inconsistent monitoring dashboards, severe alert fatigue due to duplicate or unactionable alerts, and prolonged MTTR for incidents because critical data was scattered across disparate systems.

Task

Consolidate and optimize Highspot's observability stack by executing a comprehensive migration to a unified New Relic platform. This involved deprecating all legacy systems and implementing robust, cost-efficient metric and log management strategies to drastically reduce operational costs, enhance system reliability, and significantly improve incident response capabilities.

Action

Conducted detailed analysis of existing observability costs and usage patterns

Spearheaded complete phased migration to unified New Relic platform

Deployed Prometheus-Operator with 200+ recording rules for 1000+ metrics

Implemented "Crew Ownership and Metrics Budgeting" model

Automated alert validation across New Relic, Slack, and PagerDuty

Prometheus Recording Rules for Cost Optimization

# High-cardinality optimization recording rules
groups:
- name: cost_optimization_rules
  interval: 30s
  rules:
  # Aggregate service-level metrics to reduce cardinality
  - record: service:request_rate_5m
    expr: sum(rate(http_requests_total[5m])) by (service, environment)
  
  - record: service:error_rate_5m  
    expr: |
      sum(rate(http_requests_total{status=~"5.."}[5m])) by (service, environment)
      / 
      sum(rate(http_requests_total[5m])) by (service, environment)
  
  - record: service:p99_latency_5m
    expr: histogram_quantile(0.99, sum(rate(http_request_duration_seconds_bucket[5m])) by (le, service, environment))
  
  # Infrastructure-level aggregations
  - record: node:memory_utilization_percent
    expr: (1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100
  
  - record: node:cpu_utilization_percent
    expr: 100 - (avg(irate(node_cpu_seconds_total{mode="idle"}[5m])) by (instance) * 100)
  
  # Cost attribution metrics
  - record: cluster:cost_per_cpu_hour
    expr: |
      sum(kube_node_info) by (cluster) 
      * on(cluster) group_left() 
      aws_ec2_instance_cost_per_hour
  
  - record: namespace:estimated_cost_hourly
    expr: |
      sum(
        container_spec_cpu_shares / 1024 * 
        on(instance) group_left() 
        node:cost_per_cpu_hour
      ) by (namespace, cluster)

Go-based Metrics Analysis Tool

// Go implementation for high-cardinality metric analysis
package main

import (
    "context"
    "fmt"
    "log"
    "time"
    
    "github.com/prometheus/client_golang/api"
    v1 "github.com/prometheus/client_golang/api/prometheus/v1"
    "github.com/prometheus/common/model"
)

type MetricAnalyzer struct {
    client v1.API
    config AnalysisConfig
}

type AnalysisConfig struct {
    CardinalityThreshold int
    CostThresholdUSD     float64
    RetentionDays        int
}

type MetricInsight struct {
    Name          string
    Cardinality   int
    EstimatedCost float64
    Recommendation string
}

func NewMetricAnalyzer(prometheusURL string) (*MetricAnalyzer, error) {
    client, err := api.NewClient(api.Config{
        Address: prometheusURL,
    })
    if err != nil {
        return nil, fmt.Errorf("error creating Prometheus client: %v", err)
    }
    
    return &MetricAnalyzer{
        client: v1.NewAPI(client),
        config: AnalysisConfig{
            CardinalityThreshold: 10000,
            CostThresholdUSD:     1000,
            RetentionDays:        30,
        },
    }, nil
}

func (ma *MetricAnalyzer) AnalyzeHighCardinalityMetrics(ctx context.Context) ([]MetricInsight, error) {
    // Query to find metrics with high cardinality
    query := `topk(50, count by (__name__)({__name__!=""}))`
    
    result, warnings, err := ma.client.Query(ctx, query, time.Now())
    if err != nil {
        return nil, fmt.Errorf("error querying Prometheus: %v", err)
    }
    
    if len(warnings) > 0 {
        log.Printf("Warnings: %v", warnings)
    }
    
    var insights []MetricInsight
    for _, sample := range result.(model.Vector) {
        metricName := string(sample.Metric["__name__"])
        cardinality := int(sample.Value)
        
        if cardinality > ma.config.CardinalityThreshold {
            insight := MetricInsight{
                Name:        metricName,
                Cardinality: cardinality,
                EstimatedCost: ma.calculateEstimatedCost(cardinality),
            }
            
            insight.Recommendation = ma.generateRecommendation(insight)
            insights = append(insights, insight)
        }
    }
    
    return insights, nil
}

func (ma *MetricAnalyzer) calculateEstimatedCost(cardinality int) float64 {
    // Cost calculation based on ingestion rate and storage
    // Assumes $0.15 per million samples per month (New Relic pricing model)
    samplesPerDay := float64(cardinality) * 1440 // 1 sample per minute
    samplesPerMonth := samplesPerDay * 30
    costPerMonth := (samplesPerMonth / 1000000) * 0.15
    
    return costPerMonth
}

func (ma *MetricAnalyzer) generateRecommendation(insight MetricInsight) string {
    if insight.EstimatedCost > ma.config.CostThresholdUSD {
        return fmt.Sprintf("HIGH COST: Consider aggregating or filtering this metric. Use recording rules to pre-aggregate common queries.")
    } else if insight.Cardinality > 50000 {
        return "MEDIUM RISK: Monitor growth and consider label reduction strategies."
    }
    return "LOW RISK: Continue monitoring but no immediate action needed."
}

func (ma *MetricAnalyzer) GenerateOptimizationReport() error {
    ctx := context.Background()
    
    insights, err := ma.AnalyzeHighCardinalityMetrics(ctx)
    if err != nil {
        return err
    }
    
    totalCost := 0.0
    for _, insight := range insights {
        totalCost += insight.EstimatedCost
        fmt.Printf("Metric: %s | Cardinality: %d | Est. Cost: $%.2f | %s\n", 
            insight.Name, insight.Cardinality, insight.EstimatedCost, insight.Recommendation)
    }
    
    fmt.Printf("\nTotal estimated monthly cost for high-cardinality metrics: $%.2f\n", totalCost)
    return nil
}

# EKS cluster for observability workloads
resource "aws_eks_cluster" "observability" {
  name     = "${var.environment}-observability-cluster"
  role_arn = aws_iam_role.eks_cluster_role.arn
  version  = "1.28"

  vpc_config {
    subnet_ids              = var.private_subnet_ids
    endpoint_private_access = true
    endpoint_public_access  = true
    public_access_cidrs     = ["0.0.0.0/0"]
  }

  # Enable control plane logging
  enabled_cluster_log_types = ["api", "audit", "authenticator", "controllerManager", "scheduler"]

  tags = {
    Environment = var.environment
    Purpose     = "observability-infrastructure"
    CostCenter  = "engineering"
  }
}

# Managed node group for observability workloads
resource "aws_eks_node_group" "observability_nodes" {
  cluster_name    = aws_eks_cluster.observability.name
  node_group_name = "observability-workers"
  node_role_arn   = aws_iam_role.eks_node_role.arn
  subnet_ids      = var.private_subnet_ids

  scaling_config {
    desired_size = 3
    max_size     = 10
    min_size     = 2
  }

  update_config {
    max_unavailable = 1
  }

  instance_types = ["m5.xlarge", "m5.2xlarge"]
  capacity_type  = "ON_DEMAND"

  # Disk configuration for metrics storage
  disk_size = 100

  labels = {
    role = "observability"
    environment = var.environment
  }

  tags = {
    "kubernetes.io/cluster/${aws_eks_cluster.observability.name}" = "owned"
    Environment = var.environment
  }
}

# S3 bucket for long-term metrics storage
resource "aws_s3_bucket" "metrics_storage" {
  bucket = "${var.environment}-observability-metrics-${random_id.bucket_suffix.hex}"

  tags = {
    Environment = var.environment
    Purpose     = "metrics-long-term-storage"
  }
}

resource "aws_s3_bucket_lifecycle_configuration" "metrics_lifecycle" {
  bucket = aws_s3_bucket.metrics_storage.id

  rule {
    id     = "metrics_retention"
    status = "Enabled"

    expiration {
      days = 2555  # 7 years retention
    }

    noncurrent_version_expiration {
      noncurrent_days = 30
    }
  }
}

# IAM role for Prometheus service account
resource "aws_iam_role" "prometheus_role" {
  name = "${var.environment}-prometheus-role"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Action = "sts:AssumeRoleWithWebIdentity"
        Effect = "Allow"
        Principal = {
          Federated = aws_iam_openid_connect_provider.eks.arn
        }
        Condition = {
          StringEquals = {
            "${replace(aws_iam_openid_connect_provider.eks.url, "https://", "")}:sub": "system:serviceaccount:monitoring:prometheus"
          }
        }
      }
    ]
  })
}
